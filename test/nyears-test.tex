\makeatletter
\def\@nyearsx#1/#2/#3.{\nyears{#1}{#2}{#3}}
\def\nyearsx#1{\expandafter\@nyearsx#1.}

\def\@setdatex#1/#2/#3.{
    \def\curdatex{#1/#2/#3}
    \year=#1
    \month=#2
    \day=#3
}
\def\setdatex#1{\expandafter\@setdatex#1.}

\def\printeachx#1{%
    \def\@itemsep{\par}%
    \def\nextitem{\def\nextitem{\@itemsep}}%
    \@for\el:=#1\do{\nextitem{\curdatex} $-$ {\el} $=$ {\nyearsx{\el}}}%
}

\makeatother
