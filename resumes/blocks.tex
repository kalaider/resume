\NewSetupCV{Header}
\SetupCVHeader{
    \def\Name#1{\bb{#1}\Skip{0.5\baselineskip}}%
    \def\Age{}%
}

\SetupCV{%
    \def\Layout{\MyHeader\Body}%
    \def\MyHeader{\blockHeader}%
    \def\ContactInfo{\blockContactInfo}%
    \def\Languages{\blockLanguages}%
    \def\Education{\metablockEducation}%
    \def\WorkExperience{\Block{\ENRU{Work\_ Experience}{Опыт\_ работы}}}%
    \def\Skill{\Bullet}%
    \def\Note{\Bullet[\ii{\ENRU{Note}{Замечание}}]}%
    \def\Skills#1{\Block{\ENRU{Professional\_ Skills}{Профессио\-наль\-ные\_ навыки}}{%
        {\ApplyCVSkills\Bullets{#1}}}}%
    \def\AboutMe{\BulletedBlock{\ENRU{About Me}{О себе}}}%
    \def\Other{\BulletedBlock{\ENRU{Other}{Прочее}}}%
    \def\GitHub{\Bullet[GitHub]{\url{\myGithub}}}%
    \def\GitLab{\Bullet[GitLab]{\url{\myGitlab}}}%
}

\def\blockHeader{
    \WithCVHeader{\Header{
        \Name{\myName}
        \Age{\myBirthDateAndAge}
    }}
}

\def\blockContactInfo{
    \BulletedBlock{\ENRU{Contact\_ Information}{Контактная\_ информация}}{
        \Bullet[\ENRU{Home address}{Домашний адрес}]{\myAddress}
        \Bullet[e-mail]{\email{\myEmail}}
        \Bullet[\ENRU{Phone}{Телефон}]{\myPhone}
    }
}

\def\blockLanguages{
    \Block{\ENRU{Languages}{Языки}}{
        \ENRU{Russian (native)}{Русский (родной)},
        \ENRU{Technical English (B1)}{технический английский (\en{B1})}
    }
}

\def\metablockEducation{
    \Metablock{\ENRU{Education}{Образование}}{
        \Block{\myBachelorEducationPeriod}{\myBachelorEducation}
        \Block{\myMasterEducationPeriod}{\myMasterEducation}
    }
}
